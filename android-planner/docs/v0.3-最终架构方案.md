```

└─────────────────────────────────────────┘
```

```sql
-- 备忘录
CREATE TABLE memo (
    id TEXT PRIMARY KEY,
    title TEXT,
    content TEXT,              -- 富文本 JSON

    created_at INTEGER,
    updated_at INTEGER,
    deleted_at INTEGER         -- 软删除
);

-- 日程事件
CREATE TABLE schedule_event (
    id TEXT PRIMARY KEY,
    title TEXT,
    description TEXT,
    start_time INTEGER,
    end_time INTEGER,
    all_day INTEGER DEFAULT 0,
    repeat_rule TEXT,          -- JSON: 重复规则
    reminder_config TEXT,      -- JSON: 提醒配置
    linked_task_id TEXT,       -- 关联打卡任务
    color INTEGER,
    created_at INTEGER,
    deleted_at INTEGER
);

-- 打卡任务
CREATE TABLE checkin_task (
    id TEXT PRIMARY KEY,
    title TEXT,
    description TEXT,
    group_id TEXT,
    repeat_rule TEXT,          -- JSON
    deadline_time TEXT,        -- 每日截止时间 HH:mm
    intensity_config TEXT,     -- JSON: 多级强度配置
    proof_type TEXT,           -- photo|audio|text|screenshot
    proof_config TEXT,         -- JSON: 证明要求
    current_streak INTEGER DEFAULT 0,
    best_streak INTEGER DEFAULT 0,
    total_checkins INTEGER DEFAULT 0,
    created_at INTEGER,
    archived INTEGER DEFAULT 0
);

-- 打卡记录
CREATE TABLE checkin_record (
    id TEXT PRIMARY KEY,
    task_id TEXT,
    checkin_time INTEGER,
    proof_type TEXT,
    proof_data TEXT,           -- 文件路径或文本
    ai_verified INTEGER,       -- AI 审核结果
    ai_feedback TEXT,
    status TEXT,               -- success|pending|failed
    FOREIGN KEY (task_id) REFERENCES checkin_task(id)
);

-- 密码本
CREATE TABLE password_entry (
    id TEXT PRIMARY KEY,
    title TEXT,
    group_id TEXT,
    icon TEXT,
    -- Level 0: 密码原文（字段级加密）
    password_encrypted TEXT,
    password_iv TEXT,
    -- Level 1: 提示（字段级加密，支持多条 JSON 数组）
    hints_encrypted TEXT,
    hints_iv TEXT,
    -- Level 2: 扩展字段（预留）
    ext_encrypted TEXT,
    ext_iv TEXT,
    -- 元数据（轻加密）
    metadata_encrypted TEXT,
    metadata_iv TEXT,
    created_at INTEGER,
    updated_at INTEGER,
    deleted_at INTEGER
);

-- 密码分组
CREATE TABLE password_group (
    id TEXT PRIMARY KEY,
    name TEXT,
    icon TEXT,
    color INTEGER,
    sort_order INTEGER
);

-- 灵感笔记
CREATE TABLE inspiration_note (
    id TEXT PRIMARY KEY,
    title TEXT,
    content TEXT,              -- 主文本
    template_id TEXT,
    category TEXT,             -- AI 分类
    tags TEXT,                 -- JSON 数组
    priority INTEGER DEFAULT 0,
    ai_processed INTEGER DEFAULT 0,
    ai_last_run INTEGER,
    created_at INTEGER,
    updated_at INTEGER,
    deleted_at INTEGER
);

-- 灵感附件
CREATE TABLE inspiration_attachment (
    id TEXT PRIMARY KEY,
    note_id TEXT,
    type TEXT,                 -- drawing|audio|link|web_snapshot|image
    file_path TEXT,
    url TEXT,
    metadata TEXT,             -- JSON
    created_at INTEGER,
    FOREIGN KEY (note_id) REFERENCES inspiration_note(id)
);

-- 灵感模板
CREATE TABLE inspiration_template (
    id TEXT PRIMARY KEY,
    name TEXT,
    icon TEXT,
    structure TEXT,            -- JSON: 模板字段定义
    system_prompt TEXT,        -- 关联的 AI prompt
    is_builtin INTEGER DEFAULT 0,
    sort_order INTEGER

-- AI 处理日志
CREATE TABLE ai_processing_log (
    id TEXT PRIMARY KEY,
    provider TEXT,
    model TEXT,
    input_summary TEXT,
    response_raw TEXT,
    commands_executed TEXT,    -- JSON
    status TEXT,               -- success|failed|partial
    error_message TEXT,
    tokens_used INTEGER,
    created_at INTEGER
);

-- AI Provider 配置
CREATE TABLE ai_provider_config (
    id TEXT PRIMARY KEY,
    name TEXT,
    provider_type TEXT,        -- deepseek|openai|claude|custom
    base_url TEXT,
    api_key_encrypted TEXT,    -- 字段级加密
    api_key_iv TEXT,
    model TEXT,
    system_prompt TEXT,
    max_tokens INTEGER DEFAULT 4096,
    temperature REAL DEFAULT 0.7,
    is_default INTEGER DEFAULT 0,
    created_at INTEGER
);

-- AI 定时任务
CREATE TABLE ai_scheduled_task (
    id TEXT PRIMARY KEY,
    name TEXT,
    cron_expression TEXT,
    task_type TEXT,            -- classify|summarize|organize
    target_scope TEXT,         -- JSON: 处理范围
    provider_config_id TEXT,
    enabled INTEGER DEFAULT 1,
    last_run INTEGER,
    next_run INTEGER,
    created_at INTEGER
);

-- 应用锁配置
CREATE TABLE app_lock_config (
    id TEXT PRIMARY KEY,
    lock_type TEXT,            -- app|feature
    feature_id TEXT,           -- NULL = app lock
    auth_method TEXT,          -- biometric|pin|pattern
    pin_hash TEXT,
    pin_salt TEXT,
    enabled INTEGER DEFAULT 1,
    created_at INTEGER
```

## 四、核心子系统设计

### 4.1 提醒引擎（ReminderEngine）

```
┌──────────────────────────────────────────────────┐
│                 ReminderEngine                    │
│                                                  │
│  ┌─────────────┐    ┌──────────────────┐        │
│  │ AlarmManager │    │  WorkManager     │        │
│  │ 精确闹钟      │    │  后台补偿         │        │
│  │ (唤醒设备)    │    │  (进程重启恢复)    │        │
│  └──────┬──────┘    └────────┬─────────┘        │
│         │                    │                   │
│         └────────┬───────────┘                   │
│                  ↓                               │
│         ┌─────────────────┐                     │
│         │  ReminderRouter  │                     │
│         │  (提醒路由分发)   │                     │
│         └────────┬────────┘                     │
│                  ↓                               │
│  ┌───────────────────────────────────┐          │
│  │      IntensityStrategy            │          │
│  │  ┌─────────────────────────────┐  │          │
│  │  │ Level 1: 通知栏提醒          │  │          │
│  │  │ Level 2: 重复通知 (N分钟)     │  │          │
│  │  │ Level 3: 全屏弹窗 Activity   │  │          │
│  │  │ Level 4: 锁屏通知 + 震动铃声  │  │          │
│  │  │ Level 5: 前台服务持续提醒     │  │          │
│  │  └─────────────────────────────┘  │          │
│  │                                   │          │
│  │  支持:                            │          │
│  │  - 时间触发自动升级                │          │
│  │  - 自定义各级间隔                  │          │
│  │  - 打卡验证后自动停止              │          │
│  └───────────────────────────────────┘          │
└──────────────────────────────────────────────────┘
```

**打卡提醒升级流程：**
```
任务截止前 2h  → Level 1 (通知)
任务截止前 30m → Level 2 (每5分钟重复)
任务截止前 10m → Level 3 (全屏弹窗)
截止时间到     → Level 4 (锁屏+铃声)
截止后 30m     → Level 5 (前台服务)
用户打卡+验证  → 全部停止，记录成功
```

**CommandExecutor 设计：**
```
AI 返回 → 解析 JSON → 验证命令合法性 → 执行命令 → 记录日志

支持的命令类型：
  - classify(noteId, category)     # 分类
  - tag(noteId, tags[])            # 打标签
  - summarize(noteId)              # 生成摘要
  - move(noteId, targetGroup)      # 移动分组
  - setPriority(noteId, level)     # 设置优先级
  - link(noteIdA, noteIdB)         # 建立关联
  - archive(noteId)                # 归档

每个命令支持：
  - 执行前确认（可选）
  - 执行后回滚
  - 批量执行
  - 失败重试
```

### 4.4 网页快照引擎

```
用户输入 URL
     ↓
WebView 加载（禁用 JS，纯静态）
     ↓
Document.outerHTML 获取
     ↓
图片内联为 Base64（<img src="data:...">）
     ↓
CSS 内联到 <style> 标签
     ↓
输出单个自包含 HTML 文件
     ↓
存储: /web_snapshots/{id}.html
```

**简化方案：** 不保存外部资源文件，全部内联到单 HTML。文件稍大但管理极简。
